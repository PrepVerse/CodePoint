    #include <bits/stdc++.h>
using namespace std;

void printTriangle(int n) {
    for(int i=n; i>0; i--) {
        char ch = 'A';
        for(int j=0; j<i; j++) {
            cout << ch++;
        }
        cout << endl;
    }
}

int main(void) {
    int n;
    cin >> n;
    printTriangle(n);
    return 0;
}
